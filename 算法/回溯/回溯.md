### 回溯通用模板
```javascript
function backtrack(path, choices) {
  // 结束条件
  if (满足终止条件) {
    result.push([...path]);
    return;
  }
    
  for (let i = 0; i < choices.length; i++) {
    // 剪枝
    if (无效选择) continue;
    
    // 做出选择
    path.push(choices[i]);
    
    // 递归（注意：选择列表可能需要更新）
    backtrack(path, 更新后的选择列表);
    
    // 撤销选择
    path.pop();
  }
}

```
